# renaming and moving files
git rm first.txt
git mv third.txt pudding.txt
git mv third.txt lovenotes/puddingpie.txt

# unstage a file from the staging area
git reset HEAD baddog.txt

# checkout the previous version of the file from the repository
# must not have committed bad changes yet
git checkout -- bad_dog.txt

# if bad changes were committed, and you need to checkout an older version
# 01edf8b represent first few characters of the commit ID
git checkout 01edf8b -- baddog.txt

# list all remote repos
git remote

# push your local changes to the remote repo
# remote repo named "origin"
git push origin master

# grab changes made to the remote repo
git fetch origin

# merge the changes pulled from the remote repo to your local repo
git merge origin/master

# show what branch you're currently in
git branch

# merge tool to resolve merge conflicts
git mergetool --tool=emerge

# create a new branch
git checkout -b [BRANCH_NAME]

# merge a local branch to local master
git merge [BRANCH_NAME]
